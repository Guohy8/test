#include <bits/stdc++.h>
using namespace std;

int n, m, s, mx, res;
int f[20];
bool is[20];
void dfs(int k, int ant, int val) {
    if(ant == val) {
        dfs(k + 1, 0, val);
    }
    if(k > m) {
        res = 1;
        return ;
    }
    for(int i = 1; i <= n; i++) {
        if(is[i] == false) {
            if(ant + f[i] <= val) {
                is[i] = true;
                dfs(k, ant + f[i], val);
                is[i] = false;
            }
        }
    }
}

int main () {
    cin >> n >> m;
    for(int i = 1; i <= n; i++) {
        cin >> f[i];
        s += f[i];
        mx = max(mx, f[i]);
    }
    sort(f + 1, f + n + 1);
    if(s % m != 0 or mx > s / m) {
        res = 0;
    } else {
        dfs(1, 0, s / m);
    }
    if(res == 1) {
        cout << "YES";
    } else {
        cout << "NO";
    }

    return 0;
}
